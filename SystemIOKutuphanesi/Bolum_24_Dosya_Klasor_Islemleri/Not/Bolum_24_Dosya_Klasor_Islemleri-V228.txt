        static void FileCreate(string path)
        {
            FileStream dosya = File.Create(path);
            dosya.Close();
        }

        static bool FileExists(string path)
        {
            return File.Exists(path);

        }

        static void FileDelete(string path)
        {
            File.Delete(path);

        }
